-- ১. ক্যাটাগরি টেবিল (যেমন: রাজনীতি, খেলাধুলা)
create table categories (
  id bigint generated always as identity primary key,
  name_en text not null,
  name_bn text not null,
  slug text unique not null,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- ২. জেলা টেবিল (যেমন: ঢাকা, চট্টগ্রাম)
create table districts (
  id bigint generated always as identity primary key,
  name_en text not null,
  name_bn text not null,
  slug text unique not null,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- ৩. নিউজ টেবিল
create table news (
  id bigint generated always as identity primary key,
  title text not null,       
  content text not null,    
  image_url text,             
  category_id bigint references categories(id) on delete cascade,
  district_id bigint references districts(id) on delete set null,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null,
  is_published boolean default true not null
);

drop table if exists news;
drop table if exists categories;
drop table if exists districts;

-- ৩. নিউজ টেবিল (আপনার নতুন নিয়ম অনুযায়ী)
create table news (
  id bigint generated always as identity primary key,
  title text not null,        -- রাইটার যে ভাষায় লিখবেন সেই ভাষার টাইটেল

  category_id bigint references categories(id) on delete cascade,
  district_id bigint references districts(id) on delete set null,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null,
  is_published boolean default true not null
);

-- ১. ক্যাটাগরি টেবিলের RLS পলিসি (সবাই দেখতে পারবে, সবাই যোগ করতে পারবে)
alter table categories enable row level security;

create policy "Allow public read for categories" on categories
  for select using (true);

create policy "Allow public insert for categories" on categories
  for insert with check (true);

  for delete using (true);

  ///////-- ১. রোল ট্র্যাক করার জন্য একটি নতুন টেবিল তৈরি
create table user_roles (
  id uuid references auth.users on delete cascade primary key,
  role text not null default 'user' check (role in ('admin', 'editor', 'user'))
);

-- ২. এই টেবিলটি যেন সবাই দেখতে পারে (Read-only for public checking)
alter table user_roles enable row level security;
create policy "Allow public read for user_roles" on user_roles for select using (true);

-- ৩. আপনার এডমিন ইউজারকে 'admin' রোল দেওয়া
-- (নিচের ইমেইলের জায়গায় আপনার আসল এডমিন ইমেইলটি বসিয়ে রান করবেন)
insert into user_roles (id, role)
select id, 'admin' 
from auth.users 
where email = 'admin@banshiexpress.com'
on conflict (id) do update set role = 'admin';
